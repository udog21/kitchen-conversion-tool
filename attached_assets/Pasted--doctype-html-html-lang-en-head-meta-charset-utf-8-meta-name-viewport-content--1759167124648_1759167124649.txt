  // Simple demo: swap states & adjust size via CSS variables
  const [left, right] = document.querySelectorAll('.key');
  const toggle = document.getElementById('toggle');

  toggle.addEventListener('click', () => {
    left.classList.toggle('key--up');
    left.classList.toggle('key--down');
    right.classList.toggle('key--up');
    right.classList.toggle('key--down');
  });

  const sizes = {
    sm: '72px',
    md: '96px',
    lg: '128px'
  };
  document.getElementById('sm').onclick = () => document.documentElement.style.setProperty('--key-size', sizes.sm);
  document.getElementById('md').onclick = () => document.documentElement.style.setProperty('--key-size', sizes.md);
  document.getElementById('lg').onclick = () => document.documentElement.style.setProperty('--key-size', sizes.lg);
